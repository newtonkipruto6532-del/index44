  const steps = [false, false, false];

  function openAndComplete(url, n) {
    window.open(url, '_blank');
    completeStep(n);
  }

  function completeStep(n) {
    const idx = n - 1;
    // Mark step as done after a short delay (simulating user action)
    setTimeout(() => {
      steps[idx] = true;
      const card = document.getElementById('step' + n);
      card.classList.remove('active');
      card.classList.add('done');

      // Update number to checkmark visual (done via CSS ::after on .done)
      const numEl = document.getElementById('num' + n);
      numEl.textContent = '✓';

      // Replace button with done state
      const btn = card.querySelector('.btn');
      btn.className = 'btn btn-done';
      btn.textContent = '✅ Completed';

      // Unlock next step
      if (n < 3) {
        const nextCard = document.getElementById('step' + (n + 1));
        nextCard.classList.remove('locked');
        nextCard.classList.add('active');
        showToast('✅ Step ' + n + ' done! Complete step ' + (n + 1) + ' to continue.', 'success');
      }

      // All done — show balance
      if (steps[0] && steps[1] && steps[2]) {
        setTimeout(() => {
          document.getElementById('balanceSection').classList.add('show');
          showToast('🎉 All steps complete! Your $5 BTC is ready to claim!', 'success');
        }, 600);
      }
    }, 800);
  }

  function checkBalance() {
    const wallet = document.getElementById('walletInput').value.trim();
    if (!wallet) {
      showToast('⚠️ Please enter your BTC wallet address first.', 'error');
      return;
    }
    showToast('💰 Balance confirmed: 0.000074 BTC ($5.00 USD) ✅', 'success');
  }

  function openWithdraw() {
    const wallet = document.getElementById('walletInput').value.trim();
    if (!wallet) {
      showToast('⚠️ Enter your Bitcoin wallet address to withdraw.', 'error');
      return;
    }
    document.getElementById('withdrawModal').classList.add('show');
  }

  function closeWithdraw() {
    document.getElementById('withdrawModal').classList.remove('show');
  }

  function showToast(msg, type) {
    const toast = document.getElementById('toast');
    toast.textContent = msg;
    toast.className = 'toast ' + (type || '');
    toast.classList.add('show');
    setTimeout(() => toast.classList.remove('show'), 3500);
  }

  // Close modal on overlay click
  document.getElementById('withdrawModal').addEventListener('click', function(e) {
    if (e.target === this) closeWithdraw();
  });
</script>
